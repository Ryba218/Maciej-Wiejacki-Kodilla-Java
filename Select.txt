SELECT I.SUMMARY, S.NAME 
FROM ISSUES I 
JOIN ISSUESLIST S ON S.ID = I.ISSUESLIST_ID;

SELECT I.SUMMARY, U.FIRSTNAME, U.LASTNAME 
FROM ISSUES I 
JOIN USERS U ON U.ID = I.USER_ID_ASSIGNEDTO
ORDER BY I.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS TASKS_NUMBER
FROM USERS U JOIN ISSUES I ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.LASTNAME, U.FIRSTNAME;
